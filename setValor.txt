import * as React from 'react';
import './style.css';
import { useState } from 'react';

export default function App() {
  const [valor, setValor] = useState(0);
  return (
    <div>
      <button onClick={() => setValor(valor + 1)}>
        La luz no deja de Pulsar
      </button>
      <p>{valor}</p>
    </div>
  );
}
